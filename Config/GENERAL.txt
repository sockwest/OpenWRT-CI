# -----------------------------------------------------------------
# 1. 核心代理与底层分流引擎
# -----------------------------------------------------------------

CONFIG_PACKAGE_dae=y
# 【新增修复】明确宣告使用内核原生 BTF，配合 Packages.sh 的 sed 魔改，彻底消除依赖死锁
CONFIG_DAED_USE_KERNEL_BTF=y

# [双擎代理核心] 将两大主流代理内核编译入静态文件，作为容灾备份。
CONFIG_PACKAGE_sing-box=y
CONFIG_PACKAGE_xray-core=y

# [应用层代理] 备用带图形界面的全能代理方案 (HomeProxy)。
CONFIG_PACKAGE_luci-app-homeproxy=y

# [DNS 分流防污染] 强大的 DNS 转发器，智能处理国内外 DNS 解析，防止 DNS 泄漏。
CONFIG_PACKAGE_luci-app-mosdns=y

# [异地组网主方案] Tailscale (基于 WireGuard 协议)，打通多地设备，实现无缝虚拟局域网直连。
CONFIG_PACKAGE_luci-app-tailscale=y

# [异地组网备用] VNT，另一款轻量级虚拟局域网工具，互为备份。
CONFIG_PACKAGE_luci-app-vnt=y


# -----------------------------------------------------------------
# 2. 局域网轻量化中转站与存储管理 (激活 62G eMMC 潜力)
# -----------------------------------------------------------------
# [一键分区扩容] 激活雅典娜 60G+ eMMC 闲置空间的核心工具，没有它中转站就没有物理空间。
CONFIG_PACKAGE_luci-app-partexp=y

# [轻量磁盘管理] 直观查看 eMMC 和外接 U盘 的挂载状态，支持网页端直接格式化硬盘。
CONFIG_PACKAGE_luci-app-mini-diskmanager=y

# [标准文件共享] 启用 SMB 协议，苹果生态与 Windows 原生支持最好，适合局域网视频/大物料流转。
CONFIG_PACKAGE_luci-app-samba4=y

# [极简文件快传] 网页端文件互传工具。方便手机或其他临时设备快速上传/下载工作物料。
CONFIG_PACKAGE_luci-app-quickfile=y

# [U盘/硬盘格式支持] 挂载 exFAT 和 NTFS 格式的移动硬盘 (防备偶尔插拔)。

# (强制屏蔽) 移除重度 NAS 级别的高级磁盘阵列与监控组件，避免过度占用 CPU 算力。

# -----------------------------------------------------------------

# -----------------------------------------------------------------
# 4. 实用运维与排障工具 (日常维护必备)
# -----------------------------------------------------------------
# [网页端终端] TTYD。无需打开电脑 SSH 软件，直接在浏览器内敲命令行，折腾底层协议极其方便。

CONFIG_PACKAGE_luci-app-autoreboot=y

# [MAC 流量统计] Nlbwmon。直观按局域网设备统计流量，排查某台虚拟机或新设备是否异常跑流。
CONFIG_PACKAGE_luci-app-nlbwmon=y

# [外网测速神器] Netspeedtest。直接测试物理宽带极限，排查 CPE 网络波动与节点卡顿的利器。
# 【警告】因底层 Python3 依赖冲突，当前版本已在 GENERAL 和 Packages 中强制注释屏蔽，以保全量编译通过
CONFIG_PACKAGE_luci-app-netspeedtest=y

# [内网极限测速] iperf3。精准排查局域网内的物理带宽瓶颈 (比如网线老化、网卡降速)。
CONFIG_PACKAGE_iperf3=y

# [动态域名解析] DDNS-GO。配合 IPv6，让你出门在外也能远程登录家里主路由后台管理 dae 规则。
CONFIG_PACKAGE_luci-app-ddns-go=y

# [设备时间控制] 管控特定设备的连网时段。
CONFIG_PACKAGE_luci-app-timecontrol=y

# [远程唤醒集合] 用于远程唤醒局域网内休眠的工作主机 (包含普通版和增强版)。
CONFIG_PACKAGE_luci-app-timewol=n
